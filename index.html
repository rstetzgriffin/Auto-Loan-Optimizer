import streamlit as st import pandas as pd def calculate_monthly_payment(principal, annual_rate, num_months): monthly_rate = annual_rate / 12 / 100 if monthly_rate == 0: return round(principal / num_months, 2) power = (1 + monthly_rate) ** num_months payment = principal * (monthly_rate * power) / (power - 1) return round(payment, 2) def simulate_amortization(principal, annual_rate, num_months, fixed_monthly_payment=None, lump_sum=0.0, lump_sum_month=1): monthly_rate = annual_rate / 12 / 100 monthly_payment = fixed_monthly_payment if fixed_monthly_payment is not None else calculate_monthly_payment(principal, annual_rate, num_months) balance = float(principal) total_interest = 0.0 total_paid = 0.0 actual_months = 0 for month in range(1, 1200): if balance <= 0.01: break actual_months = month interest = round(balance * monthly_rate, 2) total_interest += interest if balance + interest <= monthly_payment: payment_this_month = round(balance + interest, 2) else: payment_this_month = monthly_payment principal_paid = payment_this_month - interest balance = round(balance - principal_paid, 2) total_paid += payment_this_month if month == lump_sum_month and lump_sum > 0 and balance > 0: lump_this_month = min(lump_sum, balance) balance = round(balance - lump_this_month, 2) total_paid += lump_this_month return { 'monthly_payment': round(monthly_payment, 2), 'total_interest': round(total_interest, 2), 'actual_months': actual_months, 'total_paid': round(total_paid, 2) } # ====================== APP ====================== st.set_page_config(page_title="Auto Loan Optimizer", layout="wide") st.title("🚗 Auto Loan Strategy Optimizer") # Input fields col_input1, col_input2 = st.columns(2) with col_input1: st.subheader("Vehicle & Base Rate") vehicle_price = st.number_input("Vehicle price (out-the-door)", value=63659.0, step=100.0) base_rate = st.number_input("Base interest rate (%)", value=6.19, step=0.01) with col_input2: st.subheader("Base Terms") high_down = st.number_input("High down payment ($)", value=30000.0, step=100.0) short_term = st.number_input("Short loan term (months)", value=48, step=12) # Lump Sum Strategy inputs st.subheader("Lump Sum Strategy Options") col_strat1, col_strat2, col_strat3 = st.columns(3) with col_strat1: down_reduction = st.number_input("Reduce down payment by ($)", value=5000.0, step=100.0) low_down = high_down - down_reduction with col_strat2: long_term = st.number_input("Long loan term (months)", value=60, step=12) with col_strat3: lump_sum_rate = st.number_input("Interest rate for Lump Sum Strategy (%)", value=6.19, step=0.01) lump_sum_month = st.number_input("Apply lump sum after month", value=1, min_value=1, max_value=12) # Calculations financed_base = vehicle_price - high_down base = simulate_amortization(financed_base, base_rate, short_term) financed_extend = financed_base extended = simulate_amortization(financed_extend, base_rate, long_term) financed_lump = vehicle_price - low_down lump_sum = down_reduction lump = simulate_amortization(financed_lump, lump_sum_rate, long_term, lump_sum=lump_sum, lump_sum_month=lump_sum_month) grand_base = high_down + base['total_paid'] grand_extended = high_down + extended['total_paid'] grand_lump = low_down + lump['total_paid'] # ====================== COMPARISON TABLE ====================== st.divider() st.subheader("📊 Easy Comparison Table") data = { "Down Payment": [ f"${high_down:,.0f}", f"${high_down:,.0f}", f"${low_down:,.0f}" ], "Monthly Payment": [ f"${base['monthly_payment']:,.2f}", f"${extended['monthly_payment']:,.2f}", f"${lump['monthly_payment']:,.2f}" ], "Finance Charge": [ f"${base['total_interest']:,.2f}", f"${extended['total_interest']:,.2f}", f"${lump['total_interest']:,.2f}" ], "Months Extended": [ "—", f"+{extended['actual_months'] - base['actual_months']}", f"+{lump['actual_months'] - base['actual_months']}" ] } df = pd.DataFrame(data, index=["Base Terms", "Extended Terms", "Lump Sum Strategy"]) st.dataframe(df, use_container_width=True) # ====================== ANALYSIS SECTION ====================== st.divider() st.subheader("📝 Analysis") diff_monthly = base['monthly_payment'] - lump['monthly_payment'] diff_interest = lump['total_interest'] - base['total_interest'] extra_per_month = round(diff_interest / lump['actual_months'], 2) if lump['actual_months'] > 0 else 0 extra_months_term = lump['actual_months'] - base['actual_months'] analysis_text = f""" If you employ the **lump sum strategy**, you will save **${abs(diff_monthly):,.2f} per month**, while only incurring **${extra_per_month:,.2f} extra in financing charges per month** while extending the term of the loan by **{extra_months_term} months**. """ st.markdown(analysis_text) st.caption("💡 This strategy gives you significantly better monthly cash flow with only a small increase in the average monthly financing cost.") if st.checkbox("Show full calculation details"): st.write("**Base Terms**:", base) st.write("**Extended Terms**:", extended) st.write("**Lump Sum Strategy**:", lump)
